let xyFilter = 'all';

function buildXYGrid(filter) {
  xyGrid.innerHTML = '';
  xyData.filter(s => filter==='all' || s.group===filter).forEach(star => {
    const btn = document.createElement('div');
    btn.className = 'xy-btn';
    btn.style.borderTopColor = star.color;
    btn.dataset.name = star.name;

    const sihuaDots = star.sihua.map(sh =>
      `<span class="xy-sh-dot ${sh.type}">${{lu:'祿',quan:'權',ke:'科',ji:'忌'}[sh.type]}</span>`
    ).join('');

    btn.innerHTML = `
      <div class="xy-btn-name">${star.name}</div>
      <div class="xy-btn-wx">${star.wx.split('（')[0]}</div>
      <div class="xy-btn-sihua">${sihuaDots || '<span style="font-size:9px;color:#ccc">無四化</span>'}</div>
    `;
    btn.onclick = () => showXY(star, btn);
    xyGrid.appendChild(btn);
  });
  xyDetail.classList.remove('show');
}

function filterXY(f, btn) {
  xyFilter = f;
  document.querySelectorAll('.xy-filter-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  buildXYGrid(f);
}

function showXY(star, btn) {
  document.querySelectorAll('.xy-btn').forEach(b => b.classList.remove('selected'));
  btn.classList.add('selected');

  const sihuaHTML = star.sihua.length ? `
    <div class="xy-section">
      <div class="xy-section-title">四化解析</div>
      <div class="xy-sihua-grid">
        ${star.sihua.map(sh => `
          <div class="xy-sihua-item ${sh.type}">
            <span class="xy-sihua-label ${sh.type}">${sh.tg}年 ${{lu:'化祿',quan:'化權',ke:'化科',ji:'化忌'}[sh.type]}</span><br>
            ${sh.desc}
          </div>
        `).join('')}
      </div>
    </div>` : '';

  const healthHTML = (star.health.length || star.combo.length) ? `
    <div class="xy-section">
      <div class="xy-section-title">健康應事</div>
      <ul class="xy-section-list">
        ${star.health.map(h=>`<li>${h}</li>`).join('')}
        ${star.combo.map(c=>`<li><strong>組合：</strong>${c}</li>`).join('')}
      </ul>
    </div>` : '';

  xyDetail.innerHTML = `
    <div class="xy-detail-header">
      <div class="xy-detail-name" style="color:${star.color}">${star.name}</div>
      <div class="xy-detail-meta">
        <div class="xy-meta-row">

      <div class="xy-section-title">象義與代表</div>

  xyDetail.scrollIntoView({ behavior:'smooth', block:'nearest' });

function genComboDesc(fromGong, sihuaType, toGong) {
  const isSelf = fromGong === toGong;
  const isDui = DUI_GONG[fromGong] === toGong;
  const shName = {lu:'化祿',quan:'化權',ke:'化科',ji:'化忌'}[sihuaType];
  const shAdj = {lu:'滋潤有緣',quan:'積極壯盛',ke:'理智緩和',ji:'執著付出'}[sihuaType];
  const fromX = GONG_XIANG[fromGong];
  const toX = GONG_XIANG[toGong];
  const specialKey = `${fromGong}_${sihuaType}_${toGong}`;
  const special = SPECIAL_COMBO[specialKey];
  const detail = COMBO_DETAIL[specialKey];
  if(detail) {
    return { main:`【${detail.title}】${detail.formula}`, special:detail.note, detail, hasDetail:true };
  }
  let main = '';
  if (isSelf) {
    main = `【自化${shName}出】${fromGong}不知不覺將${shAdj}之氣散發出去，不分場合，`;
    if (sihuaType==='ji') main += '執著外露，忌出不計算忌數但需論忌出之象，可繼續追忌入對宮。';
    else if (sihuaType==='lu') main += '緣分外散，喜悅難以收攏。';
    else if (sihuaType==='quan') main += '積極外露缺乏定性，易虎頭蛇尾。';
    else main += '文質外顯易顯矯柔，優柔寡斷。';
  } else if (isDui) {
    main = `【${shName}入對宮（沖）】${fromGong}${shName}直接沖擊${toGong}，`;
    if (sihuaType==='ji') main += '忌沖對宮力道最強，為忌出形式，需繼續追忌。';
    else main += `${shAdj}之力直接衝擊，對宮承受強烈影響。`;
  } else {
    main = `【${fromGong}${shName}入${toGong}】`;
    if (sihuaType==='lu') main += `我的${fromGong}（${fromX?.wuZhi||''}）有緣滋潤${toGong}（${toX?.wuZhi||''}），帶來因緣與喜悅。`;
    else if (sihuaType==='quan') main += `我的${fromGong}積極成就${toGong}（${toX?.wuZhi||''}），使其壯盛強旺。`;
    else if (sihuaType==='ke') main += `我的${fromGong}禮教制約${toGong}，理智緩行，${toX?.wuZhi||toGong}得到細水長流的收穫。`;
    else main += `我的${fromGong}（${fromX?.wuZhi||''}）執著付出於${toGong}，${toX?.wuZhi||toGong}因此承受壓力或牽制。`;
  }
  return { main, special: special||'' };
}

function fxParse(text) {
  const result = {};
  let yearTG=null, sex=null;
  const ym = text.match(/([甲乙丙丁戊己庚辛壬癸])[子丑寅卯辰巳午未申酉戌亥]年/);
  if(ym) yearTG=ym[1];
  if(text.includes('女命')||text.includes('性别 : 女')||text.includes('性別：女')) sex='女';
  else if(text.includes('男命')||text.includes('性别 : 男')||text.includes('性別：男')) sex='男';
  const lines = text.split('\n');
  let cur=null;
  lines.forEach(line => {
    for(const [g,pat] of Object.entries(FX_GONG_PATS)) {
      if(pat.test(line)) {
        const tm = line.match(/[\[【]([甲乙丙丁戊己庚辛壬癸])([子丑寅卯辰巳午未申酉戌亥])[\]】]/);
        cur=g;
        if(!result[g]) result[g]={tg:'',stars:[]};
        if(tm) result[g].tg=tm[1];
        break;
      }
    }
    if(cur&&(line.includes('主星')||line.includes('輔星')||line.includes('主 星')||line.includes('輔 星'))) {
      FX_ALL_STARS.forEach(s=>{
        if(line.includes(s)) {
          if(!result[cur]) result[cur]={tg:'',stars:[]};
          if(!result[cur].stars.includes(s)) result[cur].stars.push(s);
        }
      });
    }
  });
  return {gong:result,yearTG,sex};
}

function fxFindGong(gongData, star) {
  for(const [g,d] of Object.entries(gongData)) {
    if(d.stars&&d.stars.some(s=>s.includes(star)||star.includes(s))) return g;
  }
  return null;
}

function fxTrace(gongData, startGong, maxDepth=6) {
  const chain=[]; let cur=startGong;
  const visited=new Set();
  for(let i=0;i<maxDepth;i++) {
    if(visited.has(cur)) break; visited.add(cur);
    const tg=gongData[cur]?.tg;
    if(!tg||!tiangan10.includes(tg)) break;
    const d=sihuaData[tg]; if(!d) break;
    const stars=gongData[cur]?.stars||[];
    const selfJi=stars.some(s=>s.includes(d.ji)||d.ji.includes(s));
    const jiTo=selfJi?cur:(fxFindGong(gongData,d.ji)||null);
    chain.push({from:cur,tg,jiStar:d.ji,to:jiTo,self:selfJi});
    if(selfJi||!jiTo) break; cur=jiTo;
  }
  return chain;
}

function jiCls(n){return n>=4?'ji4up':n>=3?'ji3':n>=2?'ji2':'ji1';}
function jiVerdict(n){
  if(n>=5) return '五忌以上：江山盡失，情勢嚴峻';
  if(n>=4) return '四忌：果報破體，丁財凋零，須極度謹慎';
  if(n>=3) return '三忌：果報傷體，大勢不妙，敗象已露';
  if(n>=2) return '二忌：破敗之始，壓力困擾浮現';
  return '一忌：守成付出，正常功課';
}

function fxLoadChart() {
  const text = document.getElementById('fxBen')?.value?.trim();
  if(!text){ alert('請先貼上命盤文字'); return; }
  const {gong, yearTG, sex} = fxParse(text);
  if(!yearTG){ alert('無法識別生年天干，請確認命盤包含如「甲辰年」等資訊'); return; }
  fxParsedGong = {gong, yearTG, sex};

  const juMatch = text.match(/([水木金土火][二三四五六])局/);
  const juMap = {'水二':2,'木三':3,'金四':4,'土五':5,'火六':6};
  const ju = juMatch ? (juMap[juMatch[1]]||5) : 5;
  fxCachedJu = ju;

  const isYangTG = ['甲','丙','戊','庚','壬'].includes(yearTG);
  const isMale = sex==='男';
  const isShun = (isYangTG&&isMale)||(!isYangTG&&!isMale);
  fxCachedIsShun = isShun;

  const mingGongDZ = text.match(/命\s*宮\[[甲乙丙丁戊己庚辛壬癸]([子丑寅卯辰巳午未申酉戌亥])\]/);
  const mingDZIdx = mingGongDZ ? DZ12.indexOf(mingGongDZ[1]) : 0;
  fxCachedMingDZIdx = mingDZIdx;

  const douJunMatch = text.match(/子年斗君\s*[:：]\s*([子丑寅卯辰巳午未申酉戌亥])/);
  fxCachedZiNianDouJun = douJunMatch ? DZ12.indexOf(douJunMatch[1]) : mingDZIdx;

  const lunarYearDZ = text.match(/農曆時間\s*:\s*[甲乙丙丁戊己庚辛壬癸]([子丑寅卯辰巳午未申酉戌亥])年/);
  const curYearDZStr = lunarYearDZ ? lunarYearDZ[1] : null;
  const chartYearM = text.match(/鐘錶時間\s*:\s*(\d{4})/);
  const chartYear = chartYearM ? parseInt(chartYearM[1]) : new Date().getFullYear();
  const birthTimeM = text.match(/鐘錶時間\s*:\s*(\d{4})-/);
  const birthYear = birthTimeM ? parseInt(birthTimeM[1]) : chartYear;
  const currentAge = chartYear - birthYear + 1;

  const daxianOrder = isShun ? SHUN_ORDER : NI_ORDER;
  const daxianGrid = document.getElementById('fxDaxianGrid');
  if(daxianGrid) {
    daxianGrid.innerHTML = '';
    for(let i=0;i<12;i++){
      const startAge = ju + i*10;
      const gongName = daxianOrder[i];
      const tg = gong[gongName]?.tg || '?';
      const btn = document.createElement('button');
      btn.className = 'fx3-ti-btn';
      btn.innerHTML = `<span style="font-size:9px;color:var(--muted)">${startAge}-${startAge+9}歲</span><br>${gongName}<span style="font-size:9px;color:var(--teal)">（${tg}）</span>`;
      btn.style.lineHeight = '1.4';
      btn.dataset.idx = i;
      btn.onclick = () => {
        document.querySelectorAll('#fxDaxianGrid .fx3-ti-btn').forEach(b=>b.classList.remove('sel'));
        btn.classList.add('sel');
        fxSelectedDaxianIdx = i;
      };
      if(currentAge >= startAge && currentAge < startAge+10){
        btn.classList.add('sel');
        fxSelectedDaxianIdx = i;
      }
      daxianGrid.appendChild(btn);
    }
  }

  const gongByDZIdxMap = {};
  SHUN_ORDER.forEach((g,i) => { gongByDZIdxMap[DZ12[(mingDZIdx+i)%12]] = g; });

  const liunianGrid = document.getElementById('fxLiunianGrid');
  if(liunianGrid) {
    liunianGrid.innerHTML = '';
    DZ12.forEach((dz, dzIdx) => {
      const douJunPos = (fxCachedZiNianDouJun + dzIdx) % 12;
      const douJunGong = gongByDZIdxMap[DZ12[douJunPos]] || '?';
      const tg = gong[douJunGong]?.tg || '?';
      const btn = document.createElement('button');
      btn.className = 'fx3-ti-btn';
      btn.innerHTML = `${dz}年<br><span style="font-size:9px;color:var(--muted)">${douJunGong}</span><span style="font-size:9px;color:var(--teal)">（${tg}）</span>`;
      btn.style.lineHeight = '1.4';
      btn.dataset.dz = dz;
      btn.onclick = () => {
        document.querySelectorAll('#fxLiunianGrid .fx3-ti-btn').forEach(b=>b.classList.remove('sel'));
        btn.classList.add('sel');
        fxSelectedLiunianDZ = dz;
      };
      if(curYearDZStr && dz===curYearDZStr){
        btn.classList.add('sel');
        fxSelectedLiunianDZ = dz;
      }
      liunianGrid.appendChild(btn);
    });
  }

  document.getElementById('fxDaxianRow').style.display='block';
  document.getElementById('fxLiunianRow').style.display='block';
  document.getElementById('fxTiGongRow').style.display='block';
}

function renderLiti() {
  const liti = litiExamples[currentLiti];
  const container = document.getElementById('litiContent');
  if(!container) return;
  container.innerHTML = `
    <div class="liti-title">${liti.title}</div>
    <div class="liti-desc">${liti.desc}</div>
    <div class="liti-chains">
      ${liti.chains.map(c=>`
        <div class="liti-chain-step">
          <div class="liti-step-num">第${c.step}忌</div>
          <div class="liti-step-content">
            <div class="liti-step-flow">${c.from}(${c.tg}) → ${c.ji}忌 → ${c.to}</div>
            <div class="liti-step-desc">${c.desc}</div>
          </div>
        </div>
      `).join('')}
    </div>
    <div class="liti-verdict">${liti.verdict}</div>
  `;
  document.querySelectorAll('.liti-tab-btn').forEach((b,i)=>{
    b.classList.toggle('active', i===currentLiti);
  });
}

function selectLiti(idx) {
  currentLiti = idx;
  renderLiti();
}

// ===== 測驗 =====
const questions = [];
Object.entries(sihuaData).forEach(([tg,d]) => {
  const types = ['lu','quan','ke','ji'];
  const labels = { lu:'化祿', quan:'化權', ke:'化科', ji:'化忌' };
  types.forEach(t => {
    const correct = d[t];
    const wrong = Object.values(sihuaData).map(x=>x[t]).filter(s=>s!==correct);
    const opts = shuffle([correct, ...wrong.slice(0,3)]);
    questions.push({ q:`天干「${tg}」的${labels[t]}是哪顆星曜？`, opts, correct, feedback:`天干${tg}：化祿${d.lu}、化權${d.quan}、化科${d.ke}、化忌${d.ji}` });
  });
});
gongData.forEach(g => {
  const wrongGongs = gongData.filter(x=>x.name!==g.name);
  const meaning = g.meanings[0];
  const opts = shuffle([g.name, ...shuffle(wrongGongs).slice(0,3).map(x=>x.name)]);
  questions.push({ q:`「${meaning.slice(0,20)}」是哪個宮位的象義？`, opts, correct:g.name, feedback:`${g.name}：${g.direct}` });
});

const shuffledQ = shuffle(questions).slice(0,20);
let quizCurrent = -1, quizScore = 0, quizAnswered = false;

function shuffle(arr) { return [...arr].sort(()=>Math.random()-0.5); }

function nextQuestion() {
  quizCurrent++;
  quizAnswered = false;
  const fb = document.getElementById('quizFeedback');
  if(fb) fb.classList.remove('show');

  if(quizCurrent >= shuffledQ.length) {
    const qDiv = document.getElementById('quizQ');
    if(qDiv) qDiv.textContent = `測驗完成！總分 ${quizScore} / ${shuffledQ.length}`;
    const optsDiv = document.getElementById('quizOpts');
    if(optsDiv) optsDiv.innerHTML = '';
    const btnNext = document.getElementById('btnNext');
    if(btnNext) { btnNext.disabled=true; btnNext.textContent='已完成'; }
    const pf = document.getElementById('progressFill');
    if(pf) pf.style.width='100%';
    return;
  }

  const q = shuffledQ[quizCurrent];
  const qDiv = document.getElementById('quizQ');
  if(qDiv) qDiv.textContent = q.q;
  const pf = document.getElementById('progressFill');
  if(pf) pf.style.width = ((quizCurrent/shuffledQ.length)*100)+'%';
  const totalEl = document.getElementById('totalNum');
  if(totalEl) totalEl.textContent = shuffledQ.length;
  const btnNext = document.getElementById('btnNext');
  if(btnNext) { btnNext.disabled=true; btnNext.textContent='下一題 →'; }

  const optsDiv = document.getElementById('quizOpts');
  if(!optsDiv) return;
  optsDiv.innerHTML = '';
  q.opts.forEach(opt => {
    const btn = document.createElement('button');
    btn.className = 'quiz-opt';
    btn.textContent = opt;
    btn.onclick = () => answerQuiz(opt, btn, q);
    optsDiv.appendChild(btn);
  });
}

function answerQuiz(choice, btn, q) {
  if(quizAnswered) return;
  quizAnswered = true;
  const isCorrect = choice === q.correct;
  if(isCorrect) { quizScore++; btn.classList.add('correct'); }
  else {
    btn.classList.add('wrong');
    document.querySelectorAll('.quiz-opt').forEach(b=>{ if(b.textContent===q.correct) b.classList.add('correct'); });
  }
  document.querySelectorAll('.quiz-opt').forEach(b=>b.disabled=true);
  const scoreEl = document.getElementById('scoreNum');
  if(scoreEl) scoreEl.textContent = quizScore;
  const fb = document.getElementById('quizFeedback');
  if(fb) { fb.textContent=(isCorrect?'✓ 正確！':'✗ 答錯了。')+'　'+q.feedback; fb.classList.add('show'); }
  const btnNext = document.getElementById('btnNext');
  if(btnNext) btnNext.disabled=false;
}

function switchTab(tabName, btn) {
  document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active'));
  document.querySelectorAll('nav button').forEach(b=>b.classList.remove('active'));
  const tabEl = document.getElementById('tab-'+tabName);
  if(tabEl) tabEl.classList.add('active');
  if(btn) btn.classList.add('active');
  else {
    const navBtns = document.querySelectorAll('nav button');
    const tabs = ['sihua','zhixing','luoji','jiadao','geju','ganqing','huaji','liti','fenxi','mingpan','lifa','xingyao','gong','quiz'];
    const idx = tabs.indexOf(tabName);
    if(idx>=0 && navBtns[idx]) navBtns[idx].classList.add('active');
  }
}
